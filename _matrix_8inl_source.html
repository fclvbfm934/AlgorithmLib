164  {

166  throw AlgLib::bad_dimension("Attempted to exponent a non-square matrix");
167  int counter = 0;
168  auto ret = identity(rows);
169  if(exp >= 0)
170  {
171  while(counter < exp)
172  {
173  counter++;
174  ret = ret * (*this);
175  }
176  return ret;
177  }
178  if(exp < 0)
179  {
180  int mag = -exp;
181  auto copyOf(*this);
182  // Now we append the identity matrix to the right of copyOf
183  auto idCols = ret.getColumnVectors();
184  for(auto col : idCols)
185  {
186  copyOf.addColumn(col);
187  }
188  copyOf = copyOf.rref();
189  for(int i = 0; i < rows; i++)

191  for(int j = 0; j < rows; j++)

194  throw inverse_zero("Attempted to take the inverse of an non-invertible Matrix");
195  }

201  while(counter < mag)

203  ret = ret * inverse;
204  counter++;
205  }

396  template <typename T>
397  Matrix<T> Matrix<T>::triangulate() const
398  {
399  auto the_rows = this->getRowVectors();
400  T zero = T(0); // Defines zero variable so we don't have to repeatedly construct
401  /* Algorithm:
402  *
403  * Start at the left of the Matrix, with the first row. Find the leftmost column that does contain a non-zero value
404  * Use that to change all entries below it to 0
405  * Move on to the next row.
406  */
407 
408  int pivot = 0; // This is meant to store the left most non-zero value in the Matrix on row r
409 
410  for(int r = 0; r < static_cast<int>(the_rows.size()); r++)
411  {
412  /* The value of r will loop through the row indexes */
413  for(int c = pivot; c < columns; c++)
414  {
415  /* Tries to see if there are any nonzero entries from [r][c] to [rows-1][c] */
416  bool allZero = true;
417  int rest;
418  for(rest = r; rest < rows; rest++)
419  {
420  if (the_rows[rest][c] != zero)
421  {
422  allZero = false;
423  break;
424  }
425  }
426 
427  /* If there is a nonzero... */
428  if(!allZero)
429  {
430  if(r != rest)
431  the_rows[r] += the_rows[rest]; // This will make the_rows[r][c] a nonzero entry
432 
433  /* Time to eliminate everything below the_rows[r][c] */
434  for(int i = r + 1; i < rows; i++)
435  {
436  the_rows[i] -= (the_rows[i][c] / the_rows[r][c]) * the_rows[r];
437  }
438  pivot = c + 1; // Put the pivot position one after the column position
439  break;
440  }
441  }
442  }
443  return Matrix<T>(the_rows);
444  }

446  template <typename T>
447  Matrix<T> Matrix<T>::rref() const
448  {
449  Matrix<T> triangle = this->triangulate();
450  auto the_rows = triangle.getRowVectors();
451 
452  /* We are going to loop from right to left. In each row, we take the left most non-zero.
453  * After that, we will make everything above that entry 0.
454  */
455  int r = rows - 1;
456  int pivot = columns - 1;
457  T zero = T(0);
458  for(r = rows - 1; r >= 0; r--)
459  {
460  int leftMostZero; // This is meant to store the left most NON-zero (oops sorry for the confusing var name
461  for(leftMostZero = 0; leftMostZero < columns; leftMostZero++)
462  {
463  if(the_rows[r][leftMostZero] != zero)
464  break;
465  }
466  // leftMostZero now holds the index of the leftMostZero.
467  // We must check if it is columns, because that means that row is completely zeros!
468  if(leftMostZero != columns)
469  {
470  // First, make the coefficient of the non-free variable a 1
471  the_rows[r] = the_rows[r] / (the_rows[r][leftMostZero]);
472  // Now, we want to make everything above the_rows[r][leftMostZero] a zero
473  for(int i = 0; i < r; i++)
474  {
475  the_rows[i] -= the_rows[r] * (the_rows[i][leftMostZero]);
476  }
477  pivot = leftMostZero;
478  }
479  }
480  return Matrix<T>(the_rows);
